# QUESTÃO 1: 3 episódios com mais views:
SELECT TEMPORADA, EPISODIO, TITULO, PUBLICO FROM `EPISODIOS` ORDER BY PUBLICO desc LIMIT 3;

 # QUESTÃO 2: Top 03 personagens que mais apareceram:
SELECT NOME, APARICOES FROM `PERSONAGENS` ORDER BY APARICOES desc limit 3;

# QUESTÃO 3: Top 03 regiões com mais casas:
SELECT COUNT(NOME), REGIAO FROM `CASAS` GROUP BY REGIAO ORDER BY COUNT(NOME) desc LIMIT 3;

# QUESTÃO 4: Top 3 episódios mais avaliados pelos usuários :
SELECT TEMPORADA, EPISODIO, TITULO, REVIEW_USUARIOS FROM `EPISODIOS` ORDER BY REVIEW_USUARIOS DESC LIMIT 3;

# QUESTÃO 5: Top 3 episódios mais avaliados pela críticos:
SELECT TEMPORADA, EPISODIO, TITULO, PUBLICO FROM `EPISODIOS` ORDER BY PUBLICO DESC LIMIT 5;

# QUESTÃO 6: Top 3 episódios mais mal avaliados:
SELECT TEMPORADA, EPISODIO, titulo, PONTUACAO FROM `EPISODIOS` ORDER BY PONTUACAO ASC LIMIT 3;

# QUESTÃO 7: Top 3 episódios mais longos:
SELECT TEMPORADA, EPISODIO, TITULO, DURACAO FROM `EPISODIOS` ORDER BY DURACAO DESC LIMIT 3;

# QUESTÃO 8: Top 9 episódios mais curtos:
SELECT TEMPORADA, EPISODIO, TITULO, DURACAO FROM `EPISODIOS` ORDER BY DURACAO asc LIMIT 9;

# QUESTÃO 9: EPS MAIS VISTOS:
SELECT TEMPORADA, EPISODIO, TITULO, PUBLICO FROM `EPISODIOS` ORDER BY PUBLICO DESC LIMIT 7;

# QUESTÃO 10: Qual diretor dirigiu mais episódios:
SELECT DISTINCT DIRETOR, count(DIRETOR) EPISODIOS_DIRIGIDOS FROM `EPISODIOS` GROUP BY DIRETOR ORDER BY count(DIRETOR) DESC LIMIT 3;
